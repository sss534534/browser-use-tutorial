# 缓存机制详解

缓存是加速重复任务的关键。browser-use 中有多层缓存可以利用：

## 1️⃣ **DOM 选择器缓存** (最关键)

### 问题
每次交互都需要重新构建 DOM 树，非常耗时。

### 解决方案

```python
from functools import lru_cache
from typing import Optional, Dict
from dataclasses import dataclass, field
import hashlib
from browser_use.dom.views import EnhancedDOMTreeNode

@dataclass
class DOMCache:
    """DOM 元素缓存系统"""
    
    # URL -> 选择器映射
    url_selector_cache: Dict[str, Dict[str, EnhancedDOMTreeNode]] = field(default_factory=dict)
    
    # 选择器指纹 (减少重复查询)
    selector_fingerprints: Dict[str, str] = field(default_factory=dict)
    
    # 最近访问的URL
    current_url: Optional[str] = None
    
    # 缓存命中统计
    hits: int = 0
    misses: int = 0
    
    def _get_dom_hash(self, dom_content: str) -> str:
        """计算DOM内容的哈希值"""
        return hashlib.md5(dom_content.encode()).hexdigest()
    
    def cache_selector(self, url: str, selector: str, element: EnhancedDOMTreeNode) -> None:
        """缓存 URL + 选择器 -> 元素映射"""
        if url not in self.url_selector_cache:
            self.url_selector_cache[url] = {}
        
        self.url_selector_cache[url][selector] = element
    
    def get_cached_element(self, url: str, selector: str) -> Optional[EnhancedDOMTreeNode]:
        """获取缓存的元素"""
        if url in self.url_selector_cache and selector in self.url_selector_cache[url]:
            self.hits += 1
            return self.url_selector_cache[url][selector]
        
        self.misses += 1
        return None
    
    def invalidate_url(self, url: str) -> None:
        """页面变化时，清除该URL的缓存"""
        if url in self.url_selector_cache:
            del self.url_selector_cache[url]
    
    def invalidate_all(self) -> None:
        """清空所有缓存"""
        self.url_selector_cache.clear()
    
    def get_hit_rate(self) -> float:
        """获取缓存命中率"""
        total = self.hits + self.misses
        return (self.hits / total * 100) if total > 0 else 0


# 集成到 BrowserSession
class CachedBrowserSession:
    """带缓存的浏览器会话"""
    
    def __init__(self, browser_session, enable_caching: bool = True):
        self.browser_session = browser_session
        self.dom_cache = DOMCache() if enable_caching else None
    
    async def get_cached_element_by_selector(
        self,
        selector: str,
        page_extraction_llm=None,
    ) -> Optional[EnhancedDOMTreeNode]:
        """获取元素，优先从缓存获取"""
        
        if not self.dom_cache:
            # 缓存未启用，直接查询
            page = await self.browser_session.must_get_current_page()
            elements = await page.get_elements_by_css_selector(selector)
            return elements[0] if elements else None
        
        current_url = self.browser_session.get_current_page_url()
        
        # 检查缓存
        cached = self.dom_cache.get_cached_element(current_url, selector)
        if cached:
            return cached
        
        # 缓存未命中，查询 DOM
        page = await self.browser_session.must_get_current_page()
        elements = await page.get_elements_by_css_selector(selector)
        
        if elements:
            element = elements[0]
            # 加入缓存
            self.dom_cache.cache_selector(current_url, selector, element)
            return element
        
        return None
    
    async def on_navigation(self, new_url: str) -> None:
        """页面切换时调用，清除旧URL的缓存"""
        if self.dom_cache:
            self.dom_cache.invalidate_url(new_url)
    
    def get_cache_stats(self) -> dict:
        """获取缓存统计"""
        if not self.dom_cache:
            return {"enabled": False}
        
        return {
            "enabled": True,
            "hits": self.dom_cache.hits,
            "misses": self.dom_cache.misses,
            "hit_rate": f"{self.dom_cache.get_hit_rate():.1f}%",
            "cached_urls": len(self.dom_cache.url_selector_cache),
        }
```

## 2️⃣ **截图缓存**

### 问题
每步都获取截图，数据量很大。

### 解决方案

```python
from dataclasses import dataclass
import hashlib
import zlib
from pathlib import Path
from datetime import datetime
import json

@dataclass
class ScreenshotFrame:
    """单个截图帧"""
    timestamp: float
    url: str
    screenshot_base64: str
    screenshot_hash: str  # MD5哈希，用于去重
    dom_hash: str  # DOM内容哈希
    file_size_bytes: int

class ScreenshotCache:
    """智能截图缓存系统"""
    
    def __init__(
        self,
        max_cache_size_mb: int = 500,
        similarity_threshold: float = 0.95,
        storage_dir: str = './screenshot_cache',
    ):
        self.max_cache_size_mb = max_cache_size_mb
        self.similarity_threshold = similarity_threshold
        self.storage_dir = Path(storage_dir)
        self.storage_dir.mkdir(parents=True, exist_ok=True)
        
        self.frames: list[ScreenshotFrame] = []
        self.current_size_mb = 0
    
    @staticmethod
    def _hash_screenshot(screenshot_base64: str) -> str:
        """计算截图的哈希值"""
        return hashlib.md5(screenshot_base64.encode()).hexdigest()
    
    def is_similar_to_last(
        self,
        screenshot_base64: str,
        dom_hash: str,
    ) -> bool:
        """检查是否与最后一张截图相似（跳过冗余截图）"""
        if not self.frames:
            return False
        
        last_frame = self.frames[-1]
        
        # 如果DOM完全相同，认为相似
        if last_frame.dom_hash == dom_hash:
            return True
        
        # 计算相似度
        similarity = self._calculate_similarity(
            screenshot_base64,
            last_frame.screenshot_base64,
        )
        
        return similarity >= self.similarity_threshold
    
    @staticmethod
    def _calculate_similarity(screenshot1: str, screenshot2: str) -> float:
        """计算两张截图的相似度（简化版）"""
        # 实际应用中可以用感知哈希(perceptual hash)
        hash1 = hashlib.md5(screenshot1.encode()).hexdigest()
        hash2 = hashlib.md5(screenshot2.encode()).hexdigest()
        
        # 对于演示，这里使用简单的对比
        # 生产环境应该使用像素级对比或perceptual hash
        return 1.0 if hash1 == hash2 else 0.5
    
    def cache_screenshot(
        self,
        screenshot_base64: str,
        url: str,
        dom_hash: str,
        timestamp: float,
    ) -> bool:
        """
        缓存截图
        
        Returns:
            True if cached, False if skipped due to similarity
        """
        # 检查是否与最后一张类似
        if self.is_similar_to_last(screenshot_base64, dom_hash):
            print(f"⏭️  Skipping similar screenshot (saved {len(screenshot_base64) / 1024:.1f} KB)")
            return False
        
        screenshot_hash = self._hash_screenshot(screenshot_base64)
        file_size = len(screenshot_base64.encode()) / (1024 * 1024)
        
        frame = ScreenshotFrame(
            timestamp=timestamp,
            url=url,
            screenshot_base64=screenshot_base64,
            screenshot_hash=screenshot_hash,
            dom_hash=dom_hash,
            file_size_bytes=file_size,
        )
        
        self.frames.append(frame)
        self.current_size_mb += file_size
        
        # 检查缓存大小
        if self.current_size_mb > self.max_cache_size_mb:
            self._evict_oldest_frames()
        
        return True
    
    def _evict_oldest_frames(self) -> None:
        """删除最旧的截图以保持缓存大小"""
        # 保留最近的 50% 的截图
        target_size = self.max_cache_size_mb * 0.5
        
        while self.current_size_mb > target_size and self.frames:
            removed = self.frames.pop(0)
            self.current_size_mb -= removed.file_size_bytes
            print(f"🗑️  Evicted screenshot from {removed.timestamp}")
    
    def get_frames_for_replay(self) -> list[str]:
        """获取所有缓存的截图用于重放"""
        return [f.screenshot_base64 for f in self.frames]
    
    def save_to_disk(self, task_id: str) -> str:
        """保存缓存到磁盘"""
        task_dir = self.storage_dir / task_id
        task_dir.mkdir(parents=True, exist_ok=True)
        
        # 压缩截图数据
        frames_data = []
        for i, frame in enumerate(self.frames):
            frame_file = task_dir / f"frame_{i:04d}.txt"
            frame_file.write_text(frame.screenshot_base64)
            frames_data.append({
                "index": i,
                "timestamp": frame.timestamp,
                "url": frame.url,
                "hash": frame.screenshot_hash,
                "dom_hash": frame.dom_hash,
            })
        
        # 保存元数据
        metadata_file = task_dir / "metadata.json"
        with open(metadata_file, 'w') as f:
            json.dump({
                "task_id": task_id,
                "created_at": datetime.now().isoformat(),
                "total_frames": len(self.frames),
                "total_size_mb": self.current_size_mb,
                "frames": frames_data,
            }, f, indent=2)
        
        return str(task_dir)
```

## 3️⃣ **操作结果缓存**

### 问题
相同的操作（如点击特定按钮）每次都重新执行。

### 解决方案

```python
from enum import Enum
from typing import Any
import pickle

class ActionType(Enum):
    CLICK = "click"
    INPUT = "input"
    NAVIGATE = "navigate"
    EXTRACT = "extract"
    SCROLL = "scroll"

@dataclass
class ActionCacheEntry:
    """操作缓存条目"""
    action_type: ActionType
    target: str  # 选择器或URL
    parameters: dict  # 输入参数
    result: Any  # 操作结果
    execution_time_ms: float
    timestamp: float
    
    def create_key(self) -> str:
        """生成缓存键"""
        import hashlib
        key_data = f"{self.action_type.value}:{self.target}:{json.dumps(self.parameters, sort_keys=True)}"
        return hashlib.md5(key_data.encode()).hexdigest()

class ActionResultCache:
    """操作结果缓存"""
    
    def __init__(self, max_entries: int = 1000, ttl_seconds: int = 3600):
        self.max_entries = max_entries
        self.ttl_seconds = ttl_seconds
        self.cache: Dict[str, ActionCacheEntry] = {}
        self.hits = 0
        self.misses = 0
    
    def get(self, action_type: ActionType, target: str, parameters: dict) -> Optional[Any]:
        """获取缓存的操作结果"""
        key = self._make_key(action_type, target, parameters)
        
        if key not in self.cache:
            self.misses += 1
            return None
        
        entry = self.cache[key]
        
        # 检查TTL
        import time
        if time.time() - entry.timestamp > self.ttl_seconds:
            del self.cache[key]
            self.misses += 1
            return None
        
        self.hits += 1
        return entry.result
    
    def set(
        self,
        action_type: ActionType,
        target: str,
        parameters: dict,
        result: Any,
        execution_time_ms: float,
    ) -> None:
        """缓存操作结果"""
        import time
        
        key = self._make_key(action_type, target, parameters)
        
        entry = ActionCacheEntry(
            action_type=action_type,
            target=target,
            parameters=parameters,
            result=result,
            execution_time_ms=execution_time_ms,
            timestamp=time.time(),
        )
        
        self.cache[key] = entry
        
        # 检查缓存大小
        if len(self.cache) > self.max_entries:
            self._evict_lru()
    
    def _evict_lru(self) -> None:
        """删除最少最近使用的条目"""
        # 按时间戳删除最旧的
        oldest_key = min(self.cache.keys(), key=lambda k: self.cache[k].timestamp)
        del self.cache[oldest_key]
    
    @staticmethod
    def _make_key(action_type: ActionType, target: str, parameters: dict) -> str:
        import hashlib
        key_data = f"{action_type.value}:{target}:{json.dumps(parameters, sort_keys=True)}"
        return hashlib.md5(key_data.encode()).hexdigest()
    
    def get_stats(self) -> dict:
        """获取缓存统计"""
        total = self.hits + self.misses
        return {
            "hits": self.hits,
            "misses": self.misses,
            "hit_rate": f"{(self.hits / total * 100) if total > 0 else 0:.1f}%",
            "cached_actions": len(self.cache),
        }
```

## 4️⃣ **LLM 响应缓存**

### 问题
相同的页面分析会调用多次 LLM。

### 解决方案

```python
from browser_use.llm.messages import BaseMessage

@dataclass
class LLMResponseCache:
    """LLM响应缓存（基于页面内容）"""
    
    page_content_hash: str
    task_description: str
    llm_response: str
    timestamp: float

class ContextAwareCache:
    """上下文感知的LLM缓存"""
    
    def __init__(self, max_entries: int = 100):
        self.max_entries = max_entries
        self.cache: Dict[str, LLMResponseCache] = {}
    
    @staticmethod
    def _hash_context(
        page_content: str,
        task_description: str,
        dom_content: str,
    ) -> str:
        """计算上下文哈希"""
        import hashlib
        combined = f"{page_content}:{task_description}:{dom_content}"
        return hashlib.sha256(combined.encode()).hexdigest()
    
    def get_cached_response(
        self,
        page_content: str,
        task_description: str,
        dom_content: str,
    ) -> Optional[str]:
        """获取缓存的LLM响应"""
        context_hash = self._hash_context(page_content, task_description, dom_content)
        
        if context_hash in self.cache:
            return self.cache[context_hash].llm_response
        
        return None
    
    def cache_response(
        self,
        page_content: str,
        task_description: str,
        dom_content: str,
        response: str,
    ) -> None:
        """缓存LLM响应"""
        import time
        
        context_hash = self._hash_context(page_content, task_description, dom_content)
        
        self.cache[context_hash] = LLMResponseCache(
            page_content_hash=context_hash,
            task_description=task_description,
            llm_response=response,
            timestamp=time.time(),
        )
        
        if len(self.cache) > self.max_entries:
            oldest_key = min(self.cache.keys(), key=lambda k: self.cache[k].timestamp)
            del self.cache[oldest_key]
```

```python
from browser_use import Agent, Browser, ChatOpenAI
import asyncio
import time

class CachedAgent:
    """支持多层缓存的Agent包装"""
    
    def __init__(
        self,
        llm: 'BaseChatModel',
        enable_dom_cache: bool = True,
        enable_screenshot_cache: bool = True,
        enable_action_cache: bool = True,
        enable_llm_cache: bool = True,
    ):
        self.llm = llm
        self.dom_cache = DOMCache() if enable_dom_cache else None
        self.screenshot_cache = ScreenshotCache() if enable_screenshot_cache else None
        self.action_cache = ActionResultCache() if enable_action_cache else None
        self.llm_cache = ContextAwareCache() if enable_llm_cache else None
    
    async def run_task_with_caching(
        self,
        task: str,
        max_steps: int = 100,
    ) -> dict:
        """执行任务，利用多层缓存"""
        
        browser = Browser()
        await browser.start()
        
        stats = {
            "task": task,
            "start_time": time.time(),
            "cache_stats": {},
        }
        
        try:
            # 包装 BrowserSession
            cached_session = CachedBrowserSession(
                browser.browser_session,
                enable_caching=self.dom_cache is not None,
            )
            
            agent = Agent(
                task=task,
                llm=self.llm,
                browser=browser,
                max_steps=max_steps,
            )
            
            # 执行任务
            history = await agent.run()
            
            stats["duration_seconds"] = time.time() - stats["start_time"]
            stats["success"] = True
            
            # 收集缓存统计
            if self.dom_cache:
                stats["cache_stats"]["dom"] = {
                    "hits": self.dom_cache.hits,
                    "misses": self.dom_cache.misses,
                    "hit_rate": f"{self.dom_cache.get_hit_rate():.1f}%",
                }
            
            if self.action_cache:
                stats["cache_stats"]["action"] = self.action_cache.get_stats()
            
            if self.screenshot_cache:
                stats["cache_stats"]["screenshot"] = {
                    "frames_cached": len(self.screenshot_cache.frames),
                    "size_mb": f"{self.screenshot_cache.current_size_mb:.1f}",
                }
            
            return stats
        
        finally:
            await browser.close()
    
    async def batch_run_with_caching(
        self,
        tasks: list[str],
        parallel_count: int = 1,
    ) -> list[dict]:
        """
        批量运行任务，共享缓存
        
        相同的任务或相似的页面操作会重用缓存
        """
        results = []
        
        for i, task in enumerate(tasks):
            print(f"\n{'='*60}")
            print(f"Task {i+1}/{len(tasks)}: {task[:50]}...")
            print(f"{'='*60}")
            
            result = await self.run_task_with_caching(task)
            results.append(result)
            
            # 打印缓存效果
            if result["cache_stats"]:
                print(f"\n📊 Cache Statistics:")
                for cache_type, stats in result["cache_stats"].items():
                    print(f"  {cache_type.upper()}: {stats}")
            
            print(f"⏱️  Duration: {result['duration_seconds']:.1f}s")
        
        return results


# 使用示例
```

## 6️⃣ **缓存失效策略**

```python
class CacheInvalidationStrategy:
    """缓存失效管理"""
    
    def __init__(self, dom_cache: DOMCache, action_cache: ActionResultCache):
        self.dom_cache = dom_cache
        self.action_cache = action_cache
    
    async def on_page_change(self, old_url: str, new_url: str) -> None:
        """页面变化时清除相关缓存"""
        print(f"🔄 Page changed: {old_url} → {new_url}")
        
        # 清除旧URL的DOM缓存
        self.dom_cache.invalidate_url(old_url)
        
        # 清除所有操作缓存（因为页面环境改变了）
        self.action_cache.cache.clear()
    
    async def on_dom_mutation(self, url: str) -> None:
        """DOM改变时清除缓存"""
        print(f"🔀 DOM mutated on {url}")
        self.dom_cache.invalidate_url(url)
    
    async def on_element_not_found(self, url: str, selector: str) -> None:
        """元素未找到时清除缓存"""
        print(f"❌ Element not found: {selector} on {url}")
        
        # 仅清除该选择器的缓存
        if url in self.dom_cache.url_selector_cache:
            self.dom_cache.url_selector_cache[url].pop(selector, None)
    
    def clear_all_caches(self) -> None:
        """完全清空所有缓存"""
        self.dom_cache.invalidate_all()
        self.action_cache.cache.clear()
        print("🗑️  All caches cleared")
```

## 关键优化建议

1. **分层缓存**: DOM > 操作结果 > 截图 > LLM响应
2. **智能失效**: 基于URL、DOM哈希、时间戳的多维失效
3. **压缩存储**: 使用 zlib 压缩大型对象
4. **监控**: 持续跟踪命中率并调整策略
