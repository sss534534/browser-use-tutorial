## 什么是增量式学习 (Incremental Learning)

**核心概念**：
- **首次执行**: 代理从零开始学习，逐步分析页面、识别元素、执行操作
- **后续执行**: 重用首次的学习成果（操作序列、DOM选择器、界面模式），跳过重复分析

这相当于 **录制-重放** 的混合策略，但更智能。

### 1️⃣ **核心数据结构**：操作路径记录

```python
from dataclasses import dataclass
from typing import Optional
from datetime import datetime
import json

@dataclass
class ElementSelector:
    """识别页面元素的多种方式"""
    css_selector: Optional[str] = None
    xpath: Optional[str] = None
    element_index: Optional[int] = None  # browser-use 的索引
    dom_xpath: Optional[str] = None
    
    def to_dict(self) -> dict:
        return {k: v for k, v in self.__dict__.items() if v is not None}

@dataclass
class ActionStep:
    """单个操作步骤"""
    action_type: str  # click, input, navigate, scroll, extract
    target_selector: Optional[ElementSelector] = None
    parameters: dict = None  # 参数（输入文本、坐标等）
    expected_url_pattern: Optional[str] = None  # 操作后的页面标识
    extracted_content_hash: Optional[str] = None  # 用于验证
    timestamp: datetime = None
    
    def __post_init__(self):
        if self.parameters is None:
            self.parameters = {}
        if self.timestamp is None:
            self.timestamp = datetime.now()

@dataclass
class TaskPath:
    """完整的任务执行路径"""
    task_id: str
    task_description: str
    website_url: str  # 目标网站
    steps: list[ActionStep]
    total_duration_seconds: float
    success: bool
    created_at: datetime = None
    
    def __post_init__(self):
        if self.created_at is None:
            self.created_at = datetime.now()
    
    def save_to_file(self, path: str) -> None:
        """序列化到JSON"""
        data = {
            'task_id': self.task_id,
            'task_description': self.task_description,
            'website_url': self.website_url,
            'total_duration_seconds': self.total_duration_seconds,
            'success': self.success,
            'created_at': self.created_at.isoformat(),
            'steps': [
                {
                    'action_type': s.action_type,
                    'target_selector': s.target_selector.to_dict() if s.target_selector else None,
                    'parameters': s.parameters,
                    'expected_url_pattern': s.expected_url_pattern,
                    'extracted_content_hash': s.extracted_content_hash,
                    'timestamp': s.timestamp.isoformat(),
                }
                for s in self.steps
            ]
        }
        with open(path, 'w') as f:
            json.dump(data, f, indent=2)
    
    @classmethod
    def load_from_file(cls, path: str) -> 'TaskPath':
        """从JSON反序列化"""
        with open(path, 'r') as f:
            data = json.load(f)
        
        steps = [
            ActionStep(
                action_type=s['action_type'],
                target_selector=ElementSelector(**s['target_selector']) if s['target_selector'] else None,
                parameters=s['parameters'],
                expected_url_pattern=s['expected_url_pattern'],
                extracted_content_hash=s['extracted_content_hash'],
                timestamp=datetime.fromisoformat(s['timestamp']),
            )
            for s in data['steps']
        ]
        return cls(
            task_id=data['task_id'],
            task_description=data['task_description'],
            website_url=data['website_url'],
            total_duration_seconds=data['total_duration_seconds'],
            success=data['success'],
            created_at=datetime.fromisoformat(data['created_at']),
            steps=steps,
        )
```

### 2️⃣ **学习系统**：记录执行过程

```python
import hashlib
from browser_use import Agent, BrowserSession
from browser_use.agent.views import AgentHistory, ActionResult

class TaskPathRecorder:
    """记录代理执行过程，学习操作序列"""
    
    def __init__(self, storage_dir: str = './task_paths'):
        self.storage_dir = storage_dir
        self.current_path: Optional[TaskPath] = None
        self.start_time: Optional[float] = None
        Path(storage_dir).mkdir(parents=True, exist_ok=True)
    
    async def record_agent_execution(
        self,
        agent: Agent,
        history: AgentHistory,
        task_id: str,
        task_description: str,
    ) -> TaskPath:
        """从Agent执行历史中提取操作序列"""
        import time
        
        duration = time.time() - self.start_time if self.start_time else 0
        
        steps = []
        for action_result in history.all_results:
            if action_result.error:
                continue  # 跳过失败的步骤

            action_data = self._extract_action_data(action_result)
            steps.append(action_data)

            task_id=task_id,
            task_description=task_description,
            website_url=agent.browser_session.get_current_page_url(),
            steps=steps,
            total_duration_seconds=duration,
            success=history.errors() is None,
        )
        
        # 保存到本地

        task_path.save_to_file(save_path)
        logger.info(f"Task path saved to {save_path}")
        
        return task_path
    
    def _extract_action_data(self, action_result: ActionResult) -> ActionStep:
        """从ActionResult提取操作数据"""
        # 这里需要与browser-use的ActionResult结构适配
        return ActionStep(
            action_type=action_result.action_type if hasattr(action_result, 'action_type') else 'unknown',
            parameters=action_result.parameters if hasattr(action_result, 'parameters') else {},
            extracted_content_hash=hashlib.md5(
                (action_result.extracted_content or '').encode()
            ).hexdigest() if action_result.extracted_content else None,
        )
    
    def find_similar_path(
        self,
        task_description: str,
        website_url: str,
        similarity_threshold: float = 0.8,
    ) -> Optional[TaskPath]:
        """查找相似的已记录任务路径"""
        from difflib import SequenceMatcher
        
        saved_paths = list(Path(self.storage_dir).glob('*.json'))
        best_match = None
        best_score = 0
        
        for path_file in saved_paths:
            try:
                saved_path = TaskPath.load_from_file(str(path_file))
                
                # 计算任务描述相似度
                desc_similarity = SequenceMatcher(
                    None, 
                    task_description, 
                    saved_path.task_description
                ).ratio()
                
                # 计算网站URL相似度
                url_similarity = SequenceMatcher(
                    None,
                    website_url,
                    saved_path.website_url
                ).ratio()
                
                combined_score = (desc_similarity + url_similarity) / 2
                
                if combined_score > best_score:
                    best_score = combined_score
                    best_match = saved_path
            
            except Exception as e:
                logger.warning(f"Failed to load {path_file}: {e}")
                continue
        
        if best_score >= similarity_threshold:
            logger.info(f"Found similar path with score {best_score:.2f}")
            return best_match
        
        return None
```

### 3️⃣ **重放系统**：快速执行已知操作

```python
class TaskPathReplayer:
    """重放已记录的操作序列"""
    
    def __init__(self, adapter: 'BrowserAdapter' = None):
        self.adapter = adapter
    
    async def replay_path(
        self,
        task_path: TaskPath,
        browser_session: BrowserSession,
        verify: bool = True,
    ) -> dict:
        """按照学习到的路径重新执行任务"""
        results = {
            'completed_steps': 0,
            'failed_at_step': None,
            'fallback_to_agent': False,
            'extracted_content': None,
        }
        
        try:
            # 导航到起始页面
            await browser_session._navigate_and_wait(
                task_path.website_url,
                browser_session.agent_focus_target_id
            )
            
            # 逐步执行操作
            for idx, step in enumerate(task_path.steps):
                try:
                    logger.info(f"Replaying step {idx + 1}/{len(task_path.steps)}: {step.action_type}")
                    
                    success = await self._execute_step(step, browser_session)
                    
                    if not success:
                        if verify:
                            logger.warning(f"Step {idx} failed verification, switching to agent mode")
                            results['fallback_to_agent'] = True
                            results['failed_at_step'] = idx
                            break
                    
                    results['completed_steps'] = idx + 1
                
                except Exception as e:
                    logger.error(f"Error executing step {idx}: {e}")
                    if verify:
                        results['fallback_to_agent'] = True
                        results['failed_at_step'] = idx
                        break
            
            return results
        
        except Exception as e:
            logger.error(f"Replay failed: {e}")
            results['fallback_to_agent'] = True
            return results
    
    async def _execute_step(
        self,
        step: ActionStep,
        browser_session: BrowserSession,
    ) -> bool:
        """执行单个步骤"""
        
        if step.action_type == 'click':
            # 使用保存的选择器查找元素
            element = await self._find_element(step.target_selector, browser_session)
            if not element:
                return False

            if not element:
                return False

            url = step.parameters.get('url')

                    url,
                    browser_session.agent_focus_target_id
```

### 4️⃣ **完整工作流**：集成到Agent

```python
import asyncio
from browser_use import Agent, Browser, ChatOpenAI
import logging
from pathlib import Path

logger = logging.getLogger(__name__)

class IncrementalLearningAgent:
    """支持增量式学习的Agent包装类"""
    
    def __init__(
        self,
        llm: 'BaseChatModel',
        storage_dir: str = './task_paths',
        enable_learning: bool = True,
        enable_replay: bool = True,
    ):
        self.llm = llm
        self.recorder = TaskPathRecorder(storage_dir)
        self.replayer = TaskPathReplayer()
        self.enable_learning = enable_learning
        self.enable_replay = enable_replay
    
    async def run_task(
        self,
        task_description: str,
        website_url: str,
        task_id: Optional[str] = None,
        max_steps: int = 100,
        use_learned_path: bool = True,
    ) -> dict:
        """执行任务，支持重放和学习"""
        
        if task_id is None:
            task_id = f"task_{int(time.time() * 1000)}"
        
        # 第一步：查找已学习的相似任务
        learned_path = None
        if use_learned_path and self.enable_replay:
            learned_path = self.recorder.find_similar_path(task_description, website_url)
        
        browser = Browser()
        await browser.start()
        
        try:
            # 第二步：如果找到相似路径，先尝试重放
            if learned_path:
                logger.info("Found learned path, attempting replay...")
                replay_results = await self.replayer.replay_path(
                    learned_path,
                    browser.browser_session,
                    verify=True,
                )
                
                if not replay_results['fallback_to_agent']:
                    logger.info(f"✅ Completed {replay_results['completed_steps']} steps via replay")
                    return {
                        'method': 'replay',
                        'task_id': task_id,
                        'results': replay_results,
                        'time_saved': learned_path.total_duration_seconds,
                    }
                
                # 如果重放失败，继续用Agent（可选：从失败点继续）
                logger.info(f"Replay failed at step {replay_results['failed_at_step']}, switching to agent")
            
            # 第三步：使用Agent执行（新任务或重放失败）
            logger.info("Executing task with Agent...")
            agent = Agent(
                task=task_description,
                llm=self.llm,
                browser=browser,
                max_steps=max_steps,
            )
            
            start_time = time.time()
            history = await agent.run()
            duration = time.time() - start_time
            
            # 第四步：记录新的执行路径（用于未来学习）
            if self.enable_learning:
                task_path = await self.recorder.record_agent_execution(
                    agent,
                    history,
                    task_id,
                    task_description,
                )
                logger.info(f"📚 Learned new path: {task_path}")
            
            return {
                'method': 'agent',
                'task_id': task_id,
                'results': history,
                'duration': duration,
            }
        
        finally:
            await browser.close()
    
    async def batch_execute(
        self,
        tasks: list[dict],  # [{'task': '...', 'website_url': '...', ...}, ...]
        parallel_count: int = 3,
    ) -> list[dict]:
        """并行执行多个任务，共享学习成果"""
        
        semaphore = asyncio.Semaphore(parallel_count)
        
        async def run_with_semaphore(task_spec):
            async with semaphore:
                return await self.run_task(**task_spec)
        
        results = await asyncio.gather(
            *[run_with_semaphore(task) for task in tasks]
        )
        
        return results
```

### 5️⃣ **使用示例**

```python
async def main():
    # 初始化支持增量式学习的Agent
    agent = IncrementalLearningAgent(
        llm=ChatOpenAI(model='gpt-4.1-mini'),
        storage_dir='./learned_paths',
    )
    
    # 相同任务，第一次执行（学习）
    result1 = await agent.run_task(
        task_description="Extract the first 3 posts from Hacker News",
        website_url="https://news.ycombinator.com",
        task_id="hn_extract_001",
    )
    print(f"First run: {result1['duration']}s")
    
    # 相同任务，第二次执行（重放）
    result2 = await agent.run_task(
        task_description="Extract the first 3 posts from Hacker News",
        website_url="https://news.ycombinator.com",
        task_id="hn_extract_002",
    )
    print(f"Second run method: {result2['method']}")
    
    # 批量任务（共享学习）
    tasks = [
        {
            "task_description": "Find Tesla stock price",
            "website_url": "https://finance.yahoo.com",
        },
        {
            "task_description": "Find Apple stock price",
            "website_url": "https://finance.yahoo.com",
        },
        {
            "task_description": "Find Google stock price",
            "website_url": "https://finance.yahoo.com",
        },
    ]
    
    results = await agent.batch_execute(tasks, parallel_count=2)
    
    # 统计加速效果
    replayed_count = sum(1 for r in results if r['method'] == 'replay')
    print(f"✅ Replayed {replayed_count}/{len(results)} tasks")

if __name__ == '__main__':
    asyncio.run(main())
```

## 核心优势对比

| 方面 | 普通执行 | 增量式学习 |
|-----|--------|----------|
| **首次运行** | 100% 时间 | 100% 时间（+学习开销） |
| **后续相同任务** | 100% 时间 | 10-20% 时间 ⚡ |
| **后续相似任务** | 100% 时间 | 40-60% 时间 ⚡ |
| **网页更新适应** | 需重新分析 | 自动降级到Agent |
| **内存开销** | 低 | 中等（缓存路径） |
| **学习曲线** | N/A | 随时间逐渐加速 |
